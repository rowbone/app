artDialog demo

普通对话框 
var d = dialog({
	title: '欢迎',
	content: '欢迎使用 artDialog 组件！'
});
d.show();
模态对话框 
var d = dialog({
	title: '欢迎',
	content: '欢迎使用 artDialog 组件！'
});
d.showModal();
传入字符串 
var d = dialog({
	title: 'This is title',
	content: 'This is content',
	width: 600
});

d.showModal();
传入元素节点 
var $elem = $('#divHidden');
$elem.removeClass('hidden');
dialog({
	title: '新增 demo',
	content: $elem,
	id: 'EF893L',
	width: 600
}).showModal();
